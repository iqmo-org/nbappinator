243def paths_to_tree(paths: List[str], delimiter: str) -> dict: 

244 """Convert flat paths to nested tree structure for D3.""" 

245 root = {"name": "root", "children": []} 

246 nodes = {"": root} 

247 

248 sorted_paths = sorted(paths) 

249 

250 for path in sorted_paths: 

251 parts = path.split(delimiter) 

252 current_path = "" 

253 

254 for i, part in enumerate(parts): 

255 parent_path = current_path 

256 current_path = delimiter.join(parts[: i + 1]) 

257 

258 if current_path not in nodes: 

259 node = {"name": part, "children": []} 

260 nodes[current_path] = node 

261 

262 parent = nodes.get(parent_path) 

263 if parent: 

264 parent["children"].append(node) 

265 else: 

266 root["children"].append(node) 

267 

268 # Clean up empty children arrays 

269 def clean_children(node): 

270 if not node.get("children"): 

271 if "children" in node: 

272 del node["children"] 

273 else: 

274 for child in node["children"]: 

275 clean_children(child) 

276 

277 for child in root["children"]: 

278 clean_children(child) 

279 

280 # If there's only one root child, use it as the root 

281 if len(root["children"]) == 1: 

282 return root["children"][0] 

283 else: 

284 # Handle multiple roots by creating a virtual root 

285 root["name"] = "/" 

286 return root
